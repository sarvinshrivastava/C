#include <stdio.h>
#include <math.h>
#include <string.h>
void main() {
    int sum = 0, k = 0, k1 = 1, n;
    printf("Enter the no. terms required in fibonic series:");
    scanf("%d", &n);
    sum = fib(k, k1, n, sum);
    printf("Sum of fibonic series up to %d is: %d", n, sum);
}
int fib(int k, int k1, int n, int sum){
    int k2;
    while(n >= 0){
        sum = sum + k + k1;
        printf("%d\n", sum);
        n--;
        k = k1; k1 = sum;
        fib(k, k1, n, sum);
    }
    return(sum);
}

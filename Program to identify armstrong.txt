#include <stdio.h>

int main() {
    int n, i, s = 0, k = n, a = n, p, d;
    scanf("%d", &n);
    while(n > 0){
        d = n % 10;
        n = n - d;
        n = n / 10;
        i = i + 1;
    }
    while(k > 0){
        p = k % 10;
        s = s + p ^ i;
        k = k - p;
        k = k / 10;
    }
    if(a == s){

    }
    else{
        printf("No. is not armstrong");
    }
    return 0;
}#include <stdio.h>

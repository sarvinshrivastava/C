#include <stdio.h>
#include <math.h>
#include <string.h>
void main() {
    int mul = 1, n, k;
    printf("Enter a no.");
    scanf("%d", &n);
    k = n;
    mul = fact(n, mul);
    printf("Factorial of %d is: %d", k, mul);
}
int fact(int n, int mul){
    while(n > 1){
        mul = mul * n;
        //printf("%d\n", mul);
        n--;
        fact(n, mul);
    }
    return(mul);
}

#include <stdio.h>
#include <math.h>
#include <string.h>
void main() {
    int num = 0, d, flag = 0;
    long int n;
    printf("Enter no. in binary form:");
    scanf("%ld", &n);
    while(n > 0){
        d = n % 10;
        if(d < 2){
            num += d * 10;
            n /= 10;
            flag = 0;
        }
        else{
            flag = 1;
            break;
        }
    }
    if(flag == 0){
        printf("Decimal form of is:%d", num);
    }
    else{
        printf("Enter a valid binary no.");
    }
}
